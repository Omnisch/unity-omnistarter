public partial class #PARTIALSCRIPTNAME#
{
    public static #PARTIALSCRIPTNAME# Instance { get; private set; }

    private bool EnsureSingleton()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return false;
        }
        else
        {
            Instance = this;
            this.DoWhen(
                () => gameObject.scene.isLoaded,
                () => DontDestroyOnLoad(gameObject));
            return true;
        }
    }

    private void Awake()
    {
        if (!EnsureSingleton())
            return;
    }
}

{
    public override void OnInspectorGUI() {
        EditorGUILayout.HelpBox(
            "This is a singleton that can be accessed by Instance. " +
            "But it also prevents other instances from creating, even over scenes.", MessageType.Info);
        EditorGUILayout.Space();

        DrawDefaultInspector();
    }
}
